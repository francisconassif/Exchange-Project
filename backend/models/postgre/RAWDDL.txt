DROP TABLE IF EXISTS countries CASCADE ;
DROP TABLE IF EXISTS type_users CASCADE ;
DROP TABLE IF EXISTS roles CASCADE;
DROP TABLE IF EXISTS user_roles CASCADE;
DROP TABLE IF EXISTS users CASCADE;
DROP TABLE IF EXISTS games CASCADE;
DROP TABLE IF EXISTS matches CASCADE;

CREATE TABLE countries
(
    country_id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    logo TEXT NOT NULL
);

/*Tipo de usuario y su relación con la empresa, desconocida, interesado, consumidor, trabajador*/
CREATE TABLE type_users
(
    type_id SERIAL PRIMARY KEY,
    relation TEXT NOT NULL
);
/* Un usuario es un potencial comprador de la empresa, nos interesa saber  que relación tiene con la empresa para motivos
   de marketing y rastreabilidad de posibles mercados, si es que tiene alguna relación con la empresa nos interesa
   saber por parte de que empresa, además que nos interesa saber su progreso a lo largo de los juegos para ver que tan efectivo
   es el juego educativo enseñando los temas*/

CREATE TABLE roles
(
    role_id SERIAL PRIMARY KEY,
    description TEXT UNIQUE CHECK(LENGTH(description)<30)
);

CREATE TABLE users
(
    user_id BIGSERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    country INT REFERENCES countries (country_id),
    email TEXT NOT NULL UNIQUE CHECK (LENGTH(email) < 100),
    password_hash TEXT NOT NULL,
    phone TEXT UNIQUE CHECK ( LENGTH(phone) < 13 ),
    type_of_user INT REFERENCES type_users(type_id),
    company TEXT,
    birthday timestamptz,
    is_active BOOLEAN DEFAULT TRUE,
    is_banned BOOLEAN DEFAULT FALSE
);


CREATE TABLE user_roles
(
    user_id BIGINT,
    role_id BIGINT,
    PRIMARY KEY (user_id,role_id),

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY  (role_id) REFERENCES roles(role_id)
);


/* Esta tabla aloja los distintos videojuegos que va a tener la plataforma para poder relacionarlos*/
CREATE TABLE games
(
    game_id SERIAL PRIMARY KEY,
    name TEXT NOT NULL UNIQUE
);

/*Tabla intermedia para users y juego
  UTILIDAD: Registrar cada juego jugado por cada jugador, relacionando de manera única cada juego,
  aunque no es tan eficiente para data intensive aplications pero para propositos de colección de data esta ok*/

CREATE TABLE  matches
(
    user_id BIGINT,
    game_id INT,
    time_start  TIMESTAMPTZ DEFAULT now(),
    /* La duración del juego es un elemento calculado*/
    date_end TIMESTAMPTZ,
    score INT NOT NULL DEFAULT 0,
    aciertos INT NOT NULL DEFAULT 0,
    power_ups INT NOT NULL DEFAULT 0,
    racha INT NOT NULL DEFAULT 0,

    PRIMARY KEY (user_id,game_id,time_start),

    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (game_id) REFERENCES games(game_id)
)
